   v <- var(x)

convMeters <- function(val, to="inch") {
    mult <- switch(to,inch=39.3701,foot=3.28084,yard=1.09361,mile=0.000621371,NA)
    if (is.na(mult)) stop("Unknown target unit of length.")
    else return(val*mult)
}
